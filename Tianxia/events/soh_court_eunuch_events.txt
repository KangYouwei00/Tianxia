	hide_window = yes
	is_triggered_only = yes
	
	trigger = {

			has_minor_title = title_councilmember_eunuch_6
			has_minor_title = title_councilmember_eunuch_7
		}
	}
	
	immediate = {
		#log = "eunuch.4 happened for [Root.GetBestName]!"

			limit = {
				liege = {
					tier = emperor

				}
			}
