########################################################
# MINOR TITLES
# Characters can have any number of these
########################################################
#	grant_limit: Max number of holders.
#	realm_in_name: Append the name of the country after the title (Queen [of Sweden])
#	allow: conditions for the character to hold the title
#	regent: This is the title for regents (not really "minor")
#	gain_effect: fires when the title is granted
#	lose_effect: fires when a title is lost if dismissal_trigger evaluates to false
#	message: should fire a message to the its owner when set
#	The other fields are all character modifiers.

#######################################
# SOFT: GRANTABLE - FULLY MODDABLE
#######################################

# Chinese Titles
title_councilmember_eunuch = {
	is_high_prio = yes
	dignity = 0.060
	grant_limit = 1
	opinion_effect = 5
	monthly_salary = 3.00
	monthly_prestige = 1.00
	is_voter = yes
	
	allowed_to_grant = {
		real_tier = emperor
		government = chinese_imperial_government
		has_dlc = "Zeus"
		has_dlc = "Jade Dragon"
		primary_title = {
			has_law = law_voting_power_1
		}
	}
	
	allowed_to_hold = {
		OR = {
			primary_title = { temporary = no }
			NOT = {
				primary_title = {
					always = yes
				}
			}
		}
		is_adult = yes
		is_female = no
		trait = eunuch
		can_be_councilmember_emperor_trigger = yes
	}

	revoke_trigger = {
		FROM = {
			primary_title = {
				NOT = { 
					AND = {
						has_law = succession_voting_power_1
						ROOT = { is_powerful_vassal = yes }
					}
				}
			}
		}
	}
	
	gain_effect = {
	}
	lose_effect = {
		opinion = { who = FROM modifier = opinion_fired_from_council }
		opinion = { who = FROM modifier = opinion_recently_fired }
		#FROM = {
		#	character_event = { id = eunuch.3 }
		#}
	}
	retire_effect = {
		#FROM = {
		#	character_event = { id = eunuch.3 }
		#}
	}

	message = yes
}

title_court_mu = {
	is_high_prio = yes
	dignity = 0.75
	opinion_effect = 5
	revoke_allowed = yes
	
	allowed_to_grant = {
		OR = {
			religion = korean_pagan
			religion = korean_pagan_reformed
		}
	}
	allowed_to_hold = {
		OR = {
			religion = korean_pagan
			religion = korean_pagan_reformed
		}
		age = 12
		OR = {
			AND = {
				is_female = yes
				religion = korean_pagan_reformed
				korean_pagan_reformed = {
					has_religion_feature = religion_matriarchal
				}
			}
			AND = {
				is_female = no
				religion = korean_pagan_reformed
				korean_pagan_reformed = {
					has_religion_feature = religion_patriarchal
				}
			}
			AND = {
				religion = korean_pagan_reformed
				korean_pagan_reformed = {
					NOR = {
						has_religion_feature = religion_matriarchal
						has_religion_feature = religion_patriarchal
					}
				}
			}
		}
		has_character_flag = can_be_court_mu
	}
