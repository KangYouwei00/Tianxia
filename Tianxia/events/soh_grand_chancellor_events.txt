### Got enough backing and founded the bloodline using the decision

# I am now the Grand Chancellor
character_event = {
	id = grandchancellor.14
	desc = EVTDESC_GRAND_CHANCELLOR_14
	picture = GFX_evt_china_golden_age
	border = GFX_event_normal_frame_intrigue
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_GRAND_CHANCELLOR_14 # Finally!
		
		set_character_flag = obj_become_grand_chancellor_successful
		
		if = {
			limit = {
				is_female = no
			}
			create_bloodline = {
				type = grand_chancellor_bloodline_intrigue
			}
		}
		else = {
			create_bloodline = {
				type = grand_chancellor_bloodline_intrigue
				inheritance = matrilineal
			}
		}
		if = {
			limit = {
				liege = {
					any_voter = {
						has_minor_title = title_grand_chancellor
					}
				}
			}
			liege = {
				set_character_flag = special_grand_chancellor_loss
				any_voter = {
					limit = {
						has_minor_title = title_grand_chancellor
					}
					remove_title = title_grand_chancellor
				}
				clr_character_flag = special_grand_chancellor_loss
			}
		}
		
		give_minor_title = title_grand_chancellor
		
		liege = {
			character_event = { id = grandchancellor.15 } # Inform liege
			any_voter = {
				limit = {
					ai = no
				}
				if = {
					limit = {
						has_opinion_modifier = {
							who = ROOT
							modifier = supporting_as_grand_chancellor
						}
					}
					character_event = { id = grandchancellor.16 } # Inform backers
				}
				else = {
					character_event = { id = grandchancellor.17 } # Inform other councillors
				}
			}
			any_realm_lord = {
				limit = {
					ai = no
					NAND = {
						is_voter = yes
						liege = {
							independent = yes
						}
					}
				}
				character_event = { id = grandchancellor.18 } # Inform the realm
			}
		}
	}
}

character_event = {
	id = grandchancellor.19
	desc = EVTDESC_GRAND_CHANCELLOR_19

	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_GRAND_CHANCELLOR_19 # Great!
		
		set_character_flag = obj_become_grand_chancellor_successful
		
		if = {
			limit = {
				is_female = no
			}
			create_bloodline = {
				type = grand_chancellor_bloodline_war
			}
		}
		else = {
			create_bloodline = {
				type = grand_chancellor_bloodline_war
				inheritance = matrilineal
			}
		}
		
		if = {
			limit = {
				liege = {
					any_voter = {
						has_minor_title = title_grand_chancellor

			}

					limit = {
						has_minor_title = title_grand_chancellor
					}
					remove_title = title_grand_chancellor
				}
				clr_character_flag = special_grand_chancellor_loss
			}
		}
		if = {
			limit = {
				is_voter = yes
			}
			give_minor_title = title_grand_chancellor
		}
		else_if = {
			limit = {
				has_education_diplomacy_trigger = yes
			}
			liege = {
				job_chancellor = {
					remove_title = job_chancellor
				}
			}
			give_job_title = job_chancellor
			give_minor_title = title_grand_chancellor
		}
		else_if = {
			limit = {
				has_education_martial_trigger = yes
			}
			liege = {
				job_marshal = {
					remove_title = job_marshal
				}
			}
			give_job_title = job_marshal
			give_minor_title = title_grand_chancellor
		}
		else_if = {
			limit = {
				has_education_stewardship_trigger = yes
			}
			liege = {
				job_treasurer = {
					remove_title = job_treasurer
				}
			}
			give_job_title = job_treasurer
			give_minor_title = title_grand_chancellor
		}
		else_if = {
			limit = {
				has_education_intrigue_trigger = yes
			}
			liege = {
				job_spymaster = {
					remove_title = job_spymaster
				}
			}
			give_job_title = job_spymaster
			give_minor_title = title_grand_chancellor
		}
		else_if = {
			limit = {
				has_education_learning_trigger = yes
			}
			liege = {
				job_spiritual = {
					remove_title = job_spiritual
				}
			}
			give_job_title = job_spiritual
			give_minor_title = title_grand_chancellor
		}
		
		liege = {
			character_event = { id = grandchancellor.20 } # Inform liege
			any_voter = {
				limit = {
					ai = no
				}
				if = {
					limit = {
						has_opinion_modifier = {
							who = ROOT
							modifier = supporting_as_grand_chancellor
						}
					}
					character_event = { id = grandchancellor.21 } # Inform backers
				}
				else = {
					character_event = { id = grandchancellor.22 } # Inform other councillors
				}
			}

				limit = {
					ai = no

							independent = yes
						}
