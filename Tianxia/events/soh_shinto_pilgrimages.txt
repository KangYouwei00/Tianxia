					factor = 0
					NOR = {
						trait = cruel
						trait = wroth
					}
				}
				character_event = { id = shinto.25 days = 2 random = 5 } # Lost Cruel
			}
			10 = {
				modifier = {
					factor = 0
					OR = {
						trait = envious
						trait = cruel
					}
				}
				character_event = { id = shinto.26 days = 2 random = 5 } # Gained Kind
			}
			50 = {
				character_event = { id = shinto.27 days = 2 random = 5 } # Nothing happened
			}
		}
		
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				trait = cruel
			}
		}
	}
		
	option = {
		name = EVTOPTB_SHINTO_23 # I'll give it a quick death...
		
		piety = -5
		
		random = {
			chance = 5
			clr_character_flag = good_pilgrimage
			set_character_flag = bad_pilgrimage
		}
		
		random_list = {
			10 = {
				character_event = { id = shinto.28 days = 2 random = 5 } # Someone berates you
			}
			10 = {
				modifier = {
					factor = 0
					NOT = {
						trait = cruel
					}
				}
				character_event = { id = shinto.29 days = 2 random = 5 } # Lose Cruel
			}
			10 = {
				modifier = {
					factor = 0
					OR = {
						trait = cruel
						trait = envious
					}
				}
				character_event = { id = shinto.30 days = 2 random = 5 } # Gain Kind
			}
			50 = {
				character_event = { id = shinto.31 days = 2 random = 5 } # Nothing of note
			}
		}
		
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				trait = cruel
			}
		}
	}
	
	option = {
		name = EVTOPTC_SHINTO_23 # I'll show it true suffering!
		
		piety = -25
		
		random = {
			chance = 25
			clr_character_flag = good_pilgrimage
			set_character_flag = bad_pilgrimage
		}
		
		if = {
			limit = {
				NOT = {
					trait = cruel
				}
			}
			character_event = { id = shinto.32 days = 2 random = 5 } # Gain Cruel!
		}
		else = {
			character_event = { id = shinto.33 days = 2 random = 5 } # Lose extra piety!
		}
		
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0

					trait = cruel
					trait = wroth
				}
			}

			modifier = {

			}
