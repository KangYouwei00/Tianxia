obj_become_paragon_of_enlightenment_gnostic = {
	type = character
	
	allow = {
		is_adult = yes
		NOT = { piety = 100 }
		is_ruler = yes
		religion_group = gnostic_group
		NOT = { trait = slow }
		NOT = { trait = imbecile }
		NOT = { trait = inbred }
		NOT = { trait = lunatic }
	}
	chance = {
		factor = 50
		
		modifier = {
			factor = 1.25
			trait = patient
		}
		modifier = {
			factor = 0.01
			NOT = { tier = king }
			NOT = { tier = emperor }
		}
		modifier = {
			factor = 10.0
			OR = {
				trait = detached_priest
				trait = martial_cleric
				trait = scholarly_theologian
				trait = mastermind_theologian
			}
		}
		modifier = {
			factor = 10.0
			trait = zealous
		}
		modifier = {
			factor = 0.01
			trait = cynical
		}
	}
	success = {
		piety = 500
	}
	abort = {
		OR = {
			trait = incapable
			trait = lunatic
		}
	}
	abort_effect = {
	}
	effect = {
		change_learning = 1
		piety = 100
		if = {
			limit = {
				has_nickname = no
			}
			give_nickname = nick_the_wise
		}
		religion_authority = {
			modifier = paragon_of_enlightenment
			years = 20
		}
		any_vassal = {
			limit = {
				is_theocracy = yes
				religion = ROOT
			}
			opinion = {
				who = ROOT
				modifier = opinion_enlightened
				months = 1200
			}
		}
	}
}
