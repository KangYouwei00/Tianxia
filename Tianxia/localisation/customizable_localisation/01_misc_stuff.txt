# Returns GetBestName or "me" depending on whether FROM is the target character

defined_text = {
	name = GetAppropriateNameOrMeFrom
	
	text = {
		localisation_key = GetBestName
		trigger = {
			FROM = {
				NOT = {
					character = THIS
				}
			}
		}
	}
	
	text = {
		localisation_key = ME
		trigger = {
			FROM = {
				character = THIS
			}
		}
	}
}
