# Provinces can now flip to Yamato culture
character_event = {
	id = SOHCULCON.2003
	hide_window = yes
	
	only_playable = yes
	
	trigger = {
		OR = {
			year = 1100
			non_historical_culture_names_in_alt_start_trigger = yes
		}
		NOT = {	has_global_flag = yamato_culture_flipping }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_global_flag = yamato_culture_flipping
	}
}
