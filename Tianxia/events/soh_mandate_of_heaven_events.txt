# on_death, transfer 50 % of current MoH to the heir if you're Chinese Imperial
character_event = {
	id = mandate_of_heaven.2
	hide_window = yes
	is_triggered_only = yes
	
	only_independent = yes
	only_playable = yes
	
	trigger = {
		real_tier = emperor
		government = chinese_imperial_government
	}
	
	immediate = {
		current_heir = {
			ROOT = {
				divide_variable = {
					which = mandate_of_heaven
					value = 2
				}
			}
			change_variable = {
				which = mandate_of_heaven
				which = ROOT
			}
			if = {
				limit = {
					ROOT = {
						has_character_modifier = mandate_of_heaven_5

							which = mandate_of_heaven
							value = 0

						which = mandate_of_heaven
						value = 2.5
					}

							modifier = mandate_of_heaven_4
							duration = -1

						add_character_modifier = {
							modifier = mandate_of_heaven_3
							duration = -1
						}
					}
				}
			}
			else_if = {
				limit = {
					ROOT = {
						has_character_modifier = mandate_of_heaven_3
					}
				}
				add_character_modifier = {
					modifier = mandate_of_heaven_3
					duration = -1
				}
			}
			else_if = {
				limit = {
					ROOT = {

					limit = {
						check_variable = {
							which = mandate_of_heaven

						which = mandate_of_heaven

						add_character_modifier = {
							modifier = mandate_of_heaven_3

# Follow-up on the above, clearing out the modifier if we're no longer Chinese Imperial

	id = mandate_of_heaven.3
	hide_window = yes
	is_triggered_only = yes
	
	only_playable = yes # Not ideal, as it doesn't remove it for courtiers that obviously don't possess the Mandate of Heaven, but much better from an optimization perspective...
	
	trigger = {
		NAND = {
			real_tier = emperor
			government = chinese_imperial_government
		}
	}
	
	immediate = {
		remove_character_modifier = mandate_of_heaven_5
		remove_character_modifier = mandate_of_heaven_4
		remove_character_modifier = mandate_of_heaven_3
		remove_character_modifier = mandate_of_heaven_2
		remove_character_modifier = mandate_of_heaven_1
	}
}

# Liberation-style revolt against a low MoH realm
# Triggered from "on_rebel_revolt"
province_event = {
	id = mandate_of_heaven.14
	desc = EVTDESC_MANDATE_OF_HEAVEN_14
	picture = GFX_evt_china_unrest
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes

	trigger = {
		has_dlc = "The Old Gods"
		has_game_rule = {
			name = mandate_of_heaven
			value = enabled
		}
		has_acceptable_chinese_culture_trigger = yes
		has_acceptable_chinese_religion_trigger = yes

		any_province_holding = {
			NOT = { holding_type = nomad }
		}
		
		owner = {
			top_liege = {
				real_tier = emperor
				government = chinese_imperial_government
				OR = {
					AND = {
						has_character_modifier = mandate_of_heaven_2 # China is held to a higher standard
						has_landed_title = e_china
					}
					has_character_modifier = mandate_of_heaven_1
				}
			}
		}
		
		county = {
			OR = {
				region = world_china
				owner = {
					top_liege = {
						primary_title = {
							de_jure_vassal_or_below = PREVPREV
						}
					}
				}
			}
		}
		NOT = { # Not if there is already an ongoing MoH revolt for the title
			owner = {
				top_liege = {
					war = yes
					any_war = {
						defender = { character = PREV }
						using_cb = mandate_of_heaven_uprising
						thirdparty_title_scope = {
							ROOT = {
								kingdom = {
									title = PREVPREV
								}
							}
						}
					}
				}
			}
		}
	}
	
	immediate = {
		create_character = {
			random_traits = yes
			dynasty = random
			religion = ROOT
			culture = ROOT
			female = no
			age = 20
			health = 7
			fertility = 0.7
			attributes = {
				diplomacy = 6
				martial = 8
				stewardship = 6
				intrigue = 2
				learning = 6
			}
			trait = just
			trait = brave
			trait = gregarious
			trait = ambitious
			trait = brilliant_strategist
			trait = inspiring_leader
			trait = peasant_leader
		}
		
		new_character = {
			set_character_flag = mandate_of_heaven_uprising
			
			create_title = {
				tier = DUKE
				landless = yes
				temporary = yes
				rebel = yes
				culture = ROOT
				name = "MOH_REVOLT"
				holder = THIS
			}
			
			wealth = 100
			
			spawn_unit = {
				province = ROOT
				home = ROOT
				owner = THIS
				#leader = THIS
				scaled_by_biggest_garrison = 1.75
				troops = {
					archers = { 6 6 }
					light_cavalry = { 3 3 }
					knights = { 1 1 }
					light_infantry = { 7 7 }
					heavy_infantry = { 3 3 }
				}
				attrition = 1.0
			}
			
			spawn_nationalist_army_effect = yes
			
			if = {
				limit = {
					has_game_rule = {
						name = provincial_revolt_strength
						value = powerful
					}
				}
				spawn_nationalist_army_effect = yes
				wealth = 150
			}
			
			if = {
				limit = {
					has_game_rule = {
						name = provincial_revolt_strength
						value = very_powerful
					}
				}
				spawn_nationalist_army_effect = yes
				spawn_nationalist_army_effect = yes
				spawn_nationalist_army_effect = yes
				wealth = 250
			}
			
			if = {
				limit = {
					has_game_rule = {
						name = provincial_revolt_strength
						value = extremely_powerful
					}
				}
				spawn_nationalist_army_effect = yes
				spawn_nationalist_army_effect = yes
				spawn_nationalist_army_effect = yes
				spawn_nationalist_army_effect = yes
				spawn_nationalist_army_effect = yes
				wealth = 350
			}
			
			# DoW on the province top liege
			ROOT = {
				owner = {
					top_liege = {
						reverse_war = {
							target = PREVPREVPREV

	only_rulers = yes
	religion = catholic

	}

				e_china = {
					has_holder = yes
					holder_scope = {
						is_chinese_emperor_trigger = yes
					}
				}
			}
			if = {
				limit = {
					e_china = {
						has_holder = yes
						holder_scope = {
							has_character_modifier = mandate_of_heaven_5
						}
					}
				}
				set_trade_route_value_multiplier = {
					which = silk_road
					value = 2
				}
			}
			else_if = {
				limit = {
					e_china = {
						has_holder = yes
						holder_scope = {
							has_character_modifier = mandate_of_heaven_4
						}
					}
				}
				set_trade_route_value_multiplier = {
					which = silk_road
					value = 1.5
				}
			}
			else_if = {
				limit = {
					e_china = {
						has_holder = yes
						holder_scope = {
							has_character_modifier = mandate_of_heaven_3
						}
					}
				}
				set_trade_route_value_multiplier = {
					which = silk_road
					value = 1
				}
			}
			else_if = {
				limit = {
					e_china = {
						has_holder = yes
						holder_scope = {
							has_character_modifier = mandate_of_heaven_2
						}
					}
				}
				set_trade_route_value_multiplier = {
					which = silk_road
					value = 0.75
				}
			}
			else_if = {
				limit = {
					e_china = {
						has_holder = yes
						holder_scope = {
							has_character_modifier = mandate_of_heaven_1
						}
					}
				}
				set_trade_route_value_multiplier = {
					which = silk_road
					value = 0.5
				}
			}
			else = { # Something weird happened...
				set_trade_route_value_multiplier = {
					which = silk_road
					value = 1
				}
			}
		}
		else = {
			set_trade_route_value_multiplier = {
				which = silk_road
				value = 1
			}
		}
	}
}
