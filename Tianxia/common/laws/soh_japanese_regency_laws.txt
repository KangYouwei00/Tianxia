# Laws controlling if the Tenno is under a regency

law_groups = {
	japanese_regency_law = {
		law_type = realm
		allowed_for_councillors = no  # The council can't vote on this!
	}
}

laws = {
	japanese_regency_law_0 = { # No regency
		group = japanese_regency_law
		
		potential = {
			holder_scope = {
				religion = shinto
				controls_religion = yes
				NOT = {
					trait = under_regency
				}
			}
		}
		
		allow = {
			always = no
		}
		
		effect = {
			holder_scope = {
				remove_trait = under_regency
				any_character = {
					limit = {
						trait = japanese_regent
					}
					remove_trait = japanese_regent
				}
			}
		}
	}
	
	japanese_regency_law_1 = { # Regency
		group = japanese_regency_law
		
		potential = {
			holder_scope = {
				religion = shinto
				controls_religion = yes
				trait = under_regency
			}
		}
		
		allow = {
			always = no
		}
		
		effect = {
			holder_scope = {
				add_trait = under_regency
			}
		}
	}
}
