# Hidden event to determine which possible Court Mu you has been discovered
character_event = {
	id = mu.2
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			religion = korean_pagan
			religion = korean_pagan_reformed
		}
		NOT = {
			any_courtier_or_vassal = {
				has_minor_title = title_court_mu

		random_list = {
			10 = {
				character_event = { id = mu.3 } # Found possible Mu - healer

			}

				character_event = { id = mu.12 } # Found possible Mu - zealot
			}

				character_event = { id = mu.13 } # Found possible Mu - drunkard
			}
			5 = {
				character_event = { id = mu.14 } # Found possible Mu - bad Learning character
			}
			10 = {
				character_event = { id = mu.15 } # Found possible Mu - okay Learning character
			}
			5 = {
				character_event = { id = mu.16 } # Found possible Mu - good Learning character
			}
			1 = {
				character_event = { id = mu.17 } # Found possible Mu - great Learning character
			}
			5 = {
				character_event = { id = mu.18 } # Found possible Mu - unremarkable character
			}
			1 = {
				modifier = {
					factor = 0
					NOT = {
						any_courtier = {
							is_landed = no
							dynasty = ROOT
							is_married = no
							is_inaccessible_or_incapable_trigger = no
							age = 12
							NOT = {
								age = 20
							}
							OR = {
								is_female = yes
								AND = {
									is_female = no
									religion = korean_pagan_reformed
									korean_pagan_reformed = {
										has_religion_feature = religion_patriarchal
									}
								}
								AND = {
									religion = korean_pagan_reformed
									korean_pagan_reformed = {
										NOR = {
											has_religion_feature = religion_matriarchal
											has_religion_feature = religion_patriarchal
										}
									}
								}
							}
							OR = {
								trait = zealous
								trait = lunatic
								trait = possessed
								trait = ambitious
								trait = erudite
							}
						}
					}
				}
				character_event = { id = mu.19 } # Found possible Mu - young relative
			}
			10 = {
				modifier = {
					factor = 0
					NOT = {
						any_vassal = {
							is_theocracy = yes
							OR = {
								religion = korean_pagan
								religion = korean_pagan_reformed
							}
							is_inaccessible_or_incapable_trigger = no
							age = 12
							OR = {

								AND = {
									is_female = no

									}

										NOR = {
											has_religion_feature = religion_matriarchal
											has_religion_feature = religion_patriarchal
										}
									}
								}
